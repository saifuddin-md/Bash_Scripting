//Manually create 3 zombie process for testing
----------------------------------------------

# nano zombie.c

#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>

int main() {
    for (int i = 0; i < 3; i++) {
        pid_t pid = fork();

        if (pid == 0) {
            // Child process exits immediately
            printf("Child %d exiting\n", i+1);
            _exit(0);
        }
    }

    // Parent does NOT call wait()
    printf("Parent sleeping...\n");
    sleep(180);

    return 0;
}

//save.
-----------------------
//run the script.
  
# gcc zombie.c -o zombie
./zombie &
